#include <iostream>
#include <cstring>
using namespace std;
class String
{
public:
    String();
	String(const char*str);
	String(const String& b);
	~String();
	String& operator =(const String& b);
	String operator +(const String& b);
	bool operator==(const String& b);
	friend ostream& operator<<(ostream& out, const String& a);
	int length();
private:
	char* data;
};

ostream& operator<<(ostream& out, const String& a)
{
	cout << a.data << endl;
	return out;
}

String::String()
{
    data = new char[1];
    *data = '\0';
}

String::String(const char* str)
{
	if (str == NULL)
	{
		data = new char[1];
		*data = '\0';
	}
	else
	{
		data = new char[strlen(str) + 1];
		strcpy(data, str);
	}
}

String::String(const String& b)
{
	data = new char[strlen(b.data) + 1];
	strcpy(data, b.data);
}

String::~String()
{
	delete[] data;
	data = NULL;
}

String& String::operator=(const String& b)
{
	if (&b == this)
	{
		return *this;
	}
	delete[] data;
	data = new char[strlen(b.data) + 1];
	strcpy(data, b.data);
	return *this;
}

String String::operator+(const String& b)
{
	String res;
	if (!data) { res = b; }
	else if (!b.data) { res = *this; }
	else
	{
		res = new char[strlen(data) + strlen(b.data) + 1];
		strcpy(res.data, data);
		strcat(res.data, b.data);
	}
	return res;
}

bool String::operator==(const String& b)
{
	if (strlen(data) != strlen(b.data))
		return false;
	return strcmp(data, b.data) ? false : true;
}

int String::length()
{
	return strlen(data);
}
