### c++ string类

```
手写string的构造，析构，拷贝构造，重载operator=,重载operator+
```
class string
{
public:
    string();
    string(const char*str);
    string(const string& b);
    ~string();
    string& operator=(const string& b);
    string& operator+(const string& b);
    bool operator==(const string& b);
    int length();
private:
    char* data;
}

string::string()
{
    data = NULL;
}

string:: string(const char* str)
{
    if (str == NULL)
    {
        data = new char[1];
        *data = '\0';
    }
    else
    {
        data = new char[strlen(str)+1];
        strcpy(data, str);
    }
}

string::string(const string& b)
{
    if (*b == this)
    {
        return *this;
    }
    delete[] data;
    data = new char[strlen(b)+1];
    strcpy(data, b.data);
    return *this;
}

string::~string()
{
    delete[] data;
    data = NULL;
}

string& string::operator=(const string& b)
{
    if (*b == this)
    {
        return *this;
    }
    delete[] data;
    data = new char[strlen(b)+1];
    strcpy(data, b.data);
    return *this
}

string& string::operator+(const string& b)
{
    string res;
    if (!data) {res = b;}
    else if (!b) {res = *this;}
    else
    {
        res = new char[strlen(data)+strlen(b.data)+1];
        strcpy(res.data, data);
        strcat(res.data, b.data);
    }
    return res;
}

bool string::operator==(const string& b)
{
    if (strlen(data) != strlen(b.data))
        return false;
    return strcmp(data, b.data)?false:true;
}

int string::length()
{
    return strlen(data);
}
